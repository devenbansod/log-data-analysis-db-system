# setup
pip install when-changed


# working explained
$ ls /var/log/audit/
.git # Initialized a local git repo here
audit.log
watcher.py

$ when-changed auditd.log "sh watcher.sh"

# watcher.sh

watcher.sh runs git diff | wc -l # `count` number of lines added since the last commit

If the `count` > N,

	git diff > input.log
	git add . && git commit -am "Save new commit"

	bash upload.sh input.log

Else,
	do nothing
